7.2 Bitverarbeitung

Quellcode: 
    import java.io.*;

    public class bitverarbeitung {

        public static void main(String args[]) {
                
            int n1 = Integer.parseInt(args[0]);
            int n2 = Integer.parseInt(args[1]);

            //carry: zu Uebertragen, shiftCarry: eigentlicher Uebertrag,
            //sum: die zu erechnende Summe 
            int carry, shiftCarry, sum;

            sum = n1 ^ n2; 
            carry = n1 & n2;

            while (carry != 0) { //addiere saemtliche Uebertraege

                shiftCarry = carry << 1;
                carry = sum & shiftCarry;
                sum = sum ^ shiftCarry; 
            }
            
            System.out.println(sum);
        }
    }
